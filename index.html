Например какой-то код

from functools import lru_cache


def moves(h):  
    a = h  
    return (a + 1), (a + 4), a*5  
  
@lru_cache(None)  
def game(m):  
    if m >= 68:  
        return 'WIN'  
  if any(game(s) == 'WIN' for s in moves(m)):  
        return 'P1'  
  if all(game(s) == 'P1' for s in moves(m)):  
        return 'B1'  
  if any(game(s) == 'B1' for s in moves(m)):  
        return 'P2'  
  if all(game(s) == 'P1' or game(s) == 'P2' for s in moves(m)):  
        return 'B1/2'  
  
for k in range(1, 68):  
    print(k, game(k))
